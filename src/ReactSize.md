#### 基本使用

```tsx
import React from 'react';
import ReactDOM from 'react-dom';
import ReactSizeable from '@zstark/react-resizable';

ReactDOM.render(<ReactSizeable />, mountNode);
```

import { IProps } from '../packages/@zstark/react-resizable/src/index.tsx.type';

renderType$(IProps, mountNode);
